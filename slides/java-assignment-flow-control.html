if/else

Two-Arm

if (thisIsTrue)
{
  //do something
}
else
{
  //do something else
}

One-Arm

if (thisIsTrue)
{
  //do something
}

if(thatIsTrue)
{
  //do something else
}

N-Arm

if (thisIsTrue)
{
  //do something
}
else if(thatIsTrue)
{
  //do something else
}
else
{
  //do another thing
}

Looping

While

int x = 7;
int i = 0;
while(i <= x);
{
  System.out.println(i);
}

For

int x = 7;
for(int i = 0; i < x; i++);
{
  System.out.println(i);
}

For

int[] nums = {1,2,3,4,5,6,7};

for(int i = 0; i < nums.length; i++);
{
  System.out.println(nums[i]);
}

Iteration

int[] nums = {1,2,3,4,5,6,7};
for(int x : nums)
{
  System.out.println(x);
}

Exceptions

int[] nums = {1,2,3,4,5,6,7};
int x = 10;

try {
  for(int i = 0; i < x; i++);
  {
    System.out.println(nums[i]);
  }
} catch (Exception e) {
  //Something wrong happened
} finally {
  //do something regardless of success or failure
}
